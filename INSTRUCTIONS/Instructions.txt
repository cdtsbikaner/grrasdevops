Welcome! This is your Docker container. Here are the instructions to use this image:
          - Step-1: Edit your /etc/hosts file, do entry of ServerName in front of this container IP
        
          - Step-2: Change entry in /etc/httpd/conf/httpd.conf file.
            Note: Configuration File is below mentioned Copy and paste with small changes:
        
          - Step-3: Change four important entry in /var/www/html/wordpress/wp-config-sample.php file according to Database, User, Password, Grant.
            Note: Configuration File is below mentioned Copy and paste with small changes:
                    
          - Step-4: Make "ServerName" in /var/www/html/wordpress/wp-config-sample.php & /etc/hosts ip entry should be same"
        
          - Step-5: Check current process running using # ps -ef & then start php-fpm socket using #/usr/sbin/php-fpm

          - Step-6: Check php-fpm socket working or not, if yes then start Apache service using #/usr/sbin/httpd
        
          - Step-7: For any info check this article: https://rakeshkumarjangid.hashnode.dev/unveiling-the-power-of-docker-building-seamless-wordpress-and-mysql-containers-on-rockylinux#heading-step-4-install-epel-amp-remi-repository

# vi /etc/httpd/conf/httpd.conf file

<VirtualHost *:80>                                     #---------->  Apache HTTP server should listen on all available IP addresses (*) on port 80.       
    ServerAdmin  jangidrakesh71@gmail.com              #-----> valid email address where the server administrator or website maintainer can be reached.          
    DocumentRoot /var/www/html/wordpress               #------> where the web files for a particular website or web application are stored.
    ServerName grras.com                               #--------> Entry should same in /etc/hosts with this domain name. 
    <Directory "/var/www/html/wordpress">              #---------> where the web files for a particular website or web application are stored.
         Allowoverride All                             
     </Directory>
    ErrorLog logs/wordpress-error_log                  #------>  error logs file related to the wordpress virtual host or directory. check /var/log/httpd.
    CustomLog logs/wordpress-access_log combined       #-------> access logs file related to the wordpress virtual host or directory. check /var/log/httpd.
</VirtualHost>
